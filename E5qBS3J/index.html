JavaScript Arrays

Arrays are a way to store multiple values in a single variable.
Arrays always start with 0 in JavaScript.

Declaration

Declare an array 
                    
                        let array = ["Item 1", "Item 2", "Item 3"];

Using console.log() with Arrays

Print the first element of an array 
                    
                        let array = ["Item 1", "Item 2", "Item 3"]; 
                        let element1 = array[0]; 

                        console.log(element1); // Outputs Item 1
Print the second element of an array 
                    
                        let array = ["Item 1", "Item 2", "Item 3"]; 
                        let element2 = array[1]; 

                        console.log(element1); // Outputs Item 2
Print the third element of an array 
                    
                        let array = ["Item 1", "Item 2", "Item 3"]; 
                        let element3 = array[2]; 

                        console.log(element1); // Outputs Item 3

Array.length

Get the length of an array (array.length) 
                    
                        let array = ["Item 1", "Item 2", "Item 3"]; 

                        console.log(array.length); // Outputs 3

Array: forEach()

Loop through an array and print each element using forEach(). You can also do other things than just console.log() 
                    
                        let array = ["Item 1", "Item 2", "Item 3"]; 

                        array.forEach(element, => { 
                            console.log(element); 
                        }); // Outputs Item 1 Item 2 Item 3

Array: push()

Add element(s) at the end of an array using push(). It returns the array's new length when put in console.log() 
                    
                        let array = ["Item 1", "Item 2", "Item 3"]; 

						console.log(array.push("Item 4")); // Outputs 4 

						array.push("Item 5", "Item 6");

						console.log(array); // Outputs ['Item 1', 'Item 2', 'Item 3', 'Item 4', 'Item 5', 'Item 6']

Array: sort()

Sort an array alphabetically (sort()) 
                    
                        let letters = ["b", "c", "a"]; 

                        console.log(letters) // Outputs ['b', 'c', 'a']

                        letters.sort();

                        console.log(letters); // Outputs ['a', 'b', 'c']

Array: reverse()

Reverse an array (reverse()) 
                    
                        let letters = ["a", "b", "c"]; 

                        console.log(letters); // Outputs ['a', 'b', 'c']

                        letters.reverse();

                        console.log(letters); // Outputs ['c', 'b', 'a']

Math.max

Find the max number from an array 
                    
                        let numbers = ["42", "69"];

                        console.log(Math.max(numbers)); // Outputs 69

                        // You can also use variables

                        let numbers2 = ["17", "29"];
                        let maxNumber = Math.max(numbers);
                        
                        console.log(maxNumber); // Outputs 29

Array: unshift()

Add element(s) at the start of an array (unshift()).
Like push(), it also returns the array's new length when put in console.log() 
					
						let array = ["Item 1", "Item 2", "Item 3"];

						console.log(array.unshift("Orange")); // Outputs 4

						array.unshift("Apple");

						console.log(array); // Outputs ['Apple', 'Orange', 'Item 1', 'Item 2', 'Item 3']

Array: pop()

Remove element(s) at the end of an array (pop()). It returns the removed element when put in console.log() 
					
						let desserts = ["Cake", "Ice Cream", "Cupcake", "Cookie"];

						console.log(desserts.pop()); // Outputs Cookie

						desserts.pop();

						console.log(desserts); // Outputs ['Cake', 'Ice Cream']

Array: concat()

Merge (join) two arrays using concat() 
					
						let games1 = ["Minecraft", "Fortnite", "Roblox"];
						let games2 = ["CS:GO", "GTA V", "League of Legends"];
		
						let games = games1.concat(games2);
		
						console.log(games); // Outputs ['Minecraft', 'Fortnite', 'Roblox', 'CS:GO', 'GTA V', 'League of Legends']

						console.log(games1.concat(games2)) // Same output as above, but different approach
